let appointments = [];
let currentTab = 'booking';

// Toast
function showToast(msg) {
  const c = document.getElementById('toast-container');
  const t = document.createElement('div');
  t.className = 'toast-enter flex items-center gap-2 px-5 py-3 rounded-lg shadow-lg text-sm font-medium';
  t.style.cssText = 'background:#8B4789;color:#fff;';
  t.innerHTML = `<span>${msg}</span>`;
  c.appendChild(t);
  setTimeout(() => { t.classList.replace('toast-enter','toast-exit'); setTimeout(() => t.remove(), 300); }, 2500);
}

// Data SDK Handler
const dataHandler = {
  onDataChanged(data) {
    appointments = data;
    updateAppointmentBadge();
    renderAppointments();
  }
};

// Initialize Data SDK
(async () => {
  const result = await window.dataSdk.init(dataHandler);
  if (!result.isOk) {
    console.error('Data SDK init failed');
  }
})();

// Services
function renderServices() {
  const g = document.getElementById('services-grid');
  g.innerHTML = services.map(s => `
    <div class="service-card card-hover">
     <div class="service-icon">${s.emoji}</div>
     <h3 style="margin: 0 0 8px 0; font-weight: 600; color: #333; font-size: 15px;">${s.name}</h3>
     <p style="margin: 0 0 12px 0; color: #999; font-size: 13px;">R$ ${s.price}</p>
    </div>
  `).join('');
}

// Booking form
function renderBookingForm() {
  const select = document.getElementById('service-select');
  select.innerHTML = '<option value="">Selecione um serviço</option>' + services.map(s => `<option value="${s.id}">${s.name} - R$ ${s.price}</option>`).join('');
  
  const today = new Date().toISOString().split('T')[0];
  document.getElementById('booking-date').setAttribute('min', today);
}

async function handleBooking(e) {
  e.preventDefault();
  
  const name = document.getElementById('client-name').value;
  const phone = document.getElementById('client-phone').value;
  const serviceId = document.getElementById('service-select').value;
  const date = document.getElementById('booking-date').value;
  const time = document.getElementById('booking-time').value;
  const notes = document.getElementById('booking-notes').value;
  
  const service = services.find(s => s.id === serviceId);
  
  const button = e.target.querySelector('button[type="submit"]');
  button.disabled = true;
  button.textContent = 'Processando...';
  
  const result = await window.dataSdk.create({
    client_name: name,
    phone: phone,
    service: service.name,
    date: date,
    time: time,
    notes: notes
  });
  
  button.disabled = false;
  button.textContent = 'Confirmar Agendamento';
  
  if (result.isOk) {
    e.target.reset();
    showToast('✓ Agendamento confirmado!');
    showTab('appointments');
  } else {
    showToast('Erro ao confirmar agendamento');
  }
}

// Appointments
function renderAppointments() {
  const c = document.getElementById('appointments-container');
  if (appointments.length === 0) {
    c.innerHTML = '<p style="color: #999; text-align: center; padding: 32px;">Nenhum agendamento ainda</p>';
    return;
  }
  
  c.innerHTML = appointments.map((a, idx) => `
    <div class="appointment-item card-hover">
     <div style="flex: 1;">
      <p style="margin: 0 0 4px 0; font-weight: 600; color: #333;">${a.client_name}</p>
      <p style="margin: 0 0 2px 0; color: #8B4789; font-weight: 500; font-size: 13px;">${a.service}</p>
      <p style="margin: 0; color: #999; font-size: 12px;">${new Date(a.date).toLocaleDateString('pt-BR')} às ${a.time}</p>
      ${a.notes ? `<p style="margin: 4px 0 0 0; color: #666; font-size: 12px;">📝 ${a.notes}</p>` : ''}
     </div>
     <button onclick="deleteAppointment(${idx})" style="background: none; border: none; cursor: pointer; color: #999;">
      <i data-lucide="trash-2" style="width:18px;height:18px;"></i>
     </button>
    </div>
  `).join('');
  lucide.createIcons({nameAttr:'data-lucide'});
}

async function deleteAppointment(idx) {
  const record = appointments[idx];
  const button = event.target.closest('button');
  button.disabled = true;
  
  const result = await window.dataSdk.delete(record);
  if (!result.isOk) {
    showToast('Erro ao deletar');
    button.disabled = false;
  } else {
    showToast('✓ Agendamento removido');
  }
}

function updateAppointmentBadge() {
  const badge = document.getElementById('appointment-badge');
  if (appointments.length > 0) {
    badge.classList.remove('hidden');
    badge.textContent = appointments.length;
  } else {
    badge.classList.add('hidden');
  }
}

// Gallery
function renderGallery() {
  const g = document.getElementById('gallery-container');
  g.innerHTML = galleryItems.map(item => `<div class="gallery-item">${item.emoji}</div>`).join('');
}

// Tabs
function showTab(tab) {
  document.querySelectorAll('[id$="-tab"]').forEach(el => el.classList.add('hidden'));
  document.getElementById(tab + '-tab').classList.remove('hidden');
  currentTab = tab;
}
